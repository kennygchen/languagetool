 package org.languagetool.tagging;
 
 import java.util.Objects;
 
 /**
  * Result of {@link WordTagger}.
  * @since 2.8
  */
 public final class TaggedWord {
 
   private final String lemma;
   private final String posTag;
 
   public TaggedWord(String lemma, String posTag) {
     this.lemma = lemma;
     this.posTag = posTag;
   }
 
   public String getLemma() {
     return lemma;
   }
 
   public String getPosTag() {
     return posTag;
   }
 
   @Override
   public String toString() {
     return lemma + "/" + posTag;
   }
 
   @Override
   public boolean equals(Object o) {
     if (this == o) return true;
     if (o == null || getClass() != o.getClass()) return false;
     TaggedWord other = (TaggedWord) o;
     return Objects.equals(lemma, other.lemma) && Objects.equals(posTag, other.posTag);
   }
 
   @Override
   public int hashCode() {
     return Objects.hash(lemma, posTag);
   }
 }
